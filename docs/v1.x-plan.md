### Store Schema 设计

```typescript
// stores/projectStore.ts
interface ProjectStore {
  currentProjectId: string | null;
  projects: Project[];
  loadProjects: () => Promise<void>;
  createProject: (name: string) => Promise<void>;
  setCurrentProject: (id: string | null) => void;
}

// stores/objectStore.ts  
interface ObjectStore {
  allObjects: WorldObject[];
  selectedObjectId: string | null;
  loadObjects: (projectId: string) => Promise<void>;
  selectObject: (id: string | null) => void;
  updateObject: (id: string, updates: Partial<WorldObject>) => Promise<void>;
  createObject: (obj: WorldObject) => Promise<void>;
}

// stores/canvasStore.ts
interface CanvasStore {
  activeCanvasTab: string;
  activeTool: CanvasTool | null;
  zoom: number;
  pan: { x: number; y: number };
  canvasStates: CanvasTabState[];
  setActiveTab: (tab: string) => void;
  setActiveTool: (tool: CanvasTool | null) => void;
  setZoom: (zoom: number) => void;
}

// stores/uiStore.ts
interface UIStore {
  activeNavTab: string;
  searchQuery: string;
  openModals: string[];
  setActiveTab: (tab: string) => void;
  setSearchQuery: (q: string) => void;
  openModal: (id: string) => void;
  closeModal: (id: string) => void;
}
```

## 风险与依赖

| 风险 | 影响 | 缓解 |
|------|------|------|
| **P1 Mutex 死锁** 未修 → Inspector 判断操作不可用 | Inspector 无法提交判断记录 | 优先修复 Mutex 死锁（优先级 P0） |
| Zustand 迁移引入回归 bug | 全部功能受影响 | 每个 store 迁移后跑回归测试（`npm run test` + `node test-alive.cjs`） |
| 画板工具与 `@xyflow/react` 版本兼容 | 连线/框选失效 | 保持 `@xyflow/react@^12` 版本 |
| 旧数据库 schema 差异大 | 迁移数据不完整 | 预研旧版 schema，做字段映射表 |
| 模板数据质量低 | 用户不用模板 | 模板内容邀请创作者填写，首批由 AI 生成 |
